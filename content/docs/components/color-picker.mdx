```tsx
import { useState } from "react";
import { ColorPicker } from "@/components/ui/color-picker";

const Page = () => {
    const [color, setColor] = useState("#4ec4b8");
    return (
        <ColorPicker value={color} onChange={(value) => setColor(value)} />
    );
}
```
