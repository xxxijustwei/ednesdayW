### Installation

<TabsContent value="cli">

```bash
npx shadcn@latest add "https://ui.ednesdayw.com/r/color-picker.json"
```

</TabsContent>

<TabsContent value="manual">

<Steps>

<Step>Install the dependencies.</Step>

```bash
npm install react-colorful
```

<Step>Install the `popover` component.</Step>

```bash
npx shadcn@latest add popover
```

<Step>Install the `button` component.</Step>

```bash
npx shadcn@latest add "https://ui.ednesdayw.com/r/button.json"
```

<Step>Install the `colors` library.</Step>

```bash
npx shadcn@latest add "https://ui.ednesdayw.com/r/colors.json"
```

<Step>Copy and paste the following code into your project's `components/ui/color-picker.tsx` file</Step>

<ComponentSource name="color-picker" />

</Steps>

</TabsContent>

</Tabs>

```tsx
import { useState } from "react";
import { ColorPicker } from "@/components/ui/color-picker";

const Page = () => {
  const [color, setColor] = useState("#4ec4b8");
  return <ColorPicker value={color} onChange={(value) => setColor(value)} />;
};
```
